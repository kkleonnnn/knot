```python
import pathlib

_PROMPT_DIR = pathlib.Path(__file__).resolve().parents[2] / "prompts"

def _load_default_prompt(name: str) -> str:
    """读 knot/prompts/{name}.md 作为默认 system prompt 文本。
```

```python
    violations = []
    for term in terms:
        for glob in INCLUDE_GLOBS:
            result = subprocess.run(

                cwd=REPO, capture_output=True, text=True,
            )
            for line in result.stdout.splitlines():
                if not line:
                    continue
                path = line.split(":", 1)[0]
                if EXCLUDE_RE.match(path):
                    continue
                violations.append(f"[{label}:{term}] {line}")
    return violations

        if len(violations) > 50:
            print(f"  ... 还有 {len(violations) - 50} 条")
```

```python
import sys
from functools import lru_cache
from cryptography.fernet import Fernet, InvalidToken

ENC_PREFIX = "enc_v1:"
_ENV = "KNOT_MASTER_KEY"


class CryptoConfigError(Exception):
    """加密配置异常 — core 层本地（守 Contract 3）。"""


class FernetAdapter:
    def __init__(self, master_key: bytes):
        self._fernet = Fernet(master_key)

    def encrypt(self, plaintext: str) -> str:
        if plaintext is None:
            return ""
        if plaintext == "":
            return ENC_PREFIX
        token = self._fernet.encrypt(plaintext.encode("utf-8")).decode("ascii")
        return ENC_PREFIX + token

    def decrypt(self, ciphertext: str) -> str:
        if not ciphertext:
            return ""
        if not ciphertext.startswith(ENC_PREFIX):
            return ciphertext
        body = ciphertext[len(ENC_PREFIX):]
        if body == "":
            return ""
        try:
            return self._fernet.decrypt(body.encode("ascii")).decode("utf-8")
        except InvalidToken as e:
            raise CryptoConfigError("解密失败 — master key 已变更或密文损坏") from e

    def is_encrypted(self, s: str) -> bool:
        return isinstance(s, str) and s.startswith(ENC_PREFIX)


def _read_master_key() -> str | None:
    """v0.6.0 单源 — 仅读 KNOT_MASTER_KEY。"""
    return os.getenv(_ENV)


def loaded_env_name() -> str | None:
    return _ENV if os.getenv(_ENV) else None


@lru_cache(maxsize=1)
def get_crypto_adapter() -> FernetAdapter:
    raw = _read_master_key()
    if not raw:
        raise CryptoConfigError(f"未设置 master key — 需配置 {_ENV}")
    try:
        key_bytes = raw.encode() if isinstance(raw, str) else raw
        return FernetAdapter(key_bytes)
    except (ValueError, TypeError) as e:
        raise CryptoConfigError(
            f"master key 格式无效（需 Fernet base64-urlsafe 32 字节）: {e}"
        ) from e


def assert_master_key_loaded() -> None:
    get_crypto_adapter()
```

```bash
# 1. 拷贝模板

{ echo ""; echo "# Generated by quick-start at $(date)"; \
  echo "KNOT_MASTER_KEY=$(python3 -c 'from cryptography.fernet import Fernet; print(Fernet.generate_key().decode())')"; \
```

> **⚠️ KNOT_MASTER_KEY 异地备份硬约束**（内测期 2 条硬约束之一）：

### J.6 Troubleshooting

#### J.6.1 `✗ KNOT 启动失败 — 缺少加密主密钥`
- 检查 .env 含 `KNOT_MASTER_KEY=<32 字节 base64-urlsafe Fernet key>`
- 检查 docker run --env-file 路径正确
- 检查 v0.4.x 残留 BIAGENT_MASTER_KEY env（v0.6.0 不读旧名）

#### J.6.2 admin 登录失败
- 默认账号 `admin / admin123`（E-9 方案 A LOCKED）
- 如已改密码忘记 → admin 后台找其他 admin 重置 / 或 sqlite3 直改 password_hash

#### J.6.3 SQL agent 走空 prompt
- `docker exec knot ls /app/knot/prompts/` 应见 sql_planner.md / clarifier.md / presenter.md
- 缺失 → Dockerfile COPY 漏 → R-PA-7.1 守护失败 → 返工 F18.4
- `docker exec knot sqlite3 /app/knot/data/knot.db "SELECT agent, length(content) FROM prompts;"` 应见 3 行非零
- 空则 `prompt_service.seed_defaults_from_files()` 没跑 → 检查 main.py startup 调用

#### J.6.4 业务 DB 连接失败
- `docker exec knot ping <db-host>` 验证网络
- admin 后台「数据源」host/port/credentials 正确
- DB 用户 SELECT 权限 + SHOW GRANTS 探测

#### J.6.5 `anyio threadpool ... saturated`
- 加 env `ANYIO_TOKENS=64` + `docker restart knot`
