## 3. AsyncLLMAdapter 实现（R-31 完整性）

### 3.1 4 个 adapter 加 async 方法

```python
# adapters/llm/anthropic_native.py
class AnthropicAdapter:
    def complete(self, req): ...    # 现有 sync 保留（R-24）

    async def acomplete(self, req: LLMRequest) -> LLMResponse:
        """v0.4.4 真异步 — anthropic.AsyncAnthropic（SDK 自带，0 新依赖）。"""
        from anthropic import AsyncAnthropic
        kwargs = {"api_key": req.api_key}
        if req.base_url:
            kwargs["base_url"] = req.base_url
        client = AsyncAnthropic(**kwargs)
        system_field = (
            [{"type": "text", "text": req.system, "cache_control": {"type": "ephemeral"}}]
            if req.enable_prompt_cache else req.system
        )
        resp = await client.messages.create(
            model=req.model_key, max_tokens=req.max_tokens,
            temperature=req.temperature, system=system_field, messages=req.messages,
        )
        return LLMResponse(
            text=resp.content[0].text,
            input_tokens=resp.usage.input_tokens,
            output_tokens=resp.usage.output_tokens,
        )

# adapters/llm/openai_compat.py
class OpenAICompatAdapter:
    async def acomplete(self, req):
        from openai import AsyncOpenAI
        client = AsyncOpenAI(api_key=req.api_key, base_url=req.base_url or None)
        resp = await client.chat.completions.create(
            model=req.model_key, max_tokens=req.max_tokens,
            temperature=req.temperature,
            messages=[{"role": "system", "content": req.system}] + req.messages,
        )
        return LLMResponse(
            text=resp.choices[0].message.content or "",
            input_tokens=resp.usage.prompt_tokens if resp.usage else 0,
            output_tokens=resp.usage.completion_tokens if resp.usage else 0,
        )

# adapters/llm/openrouter.py 同 openai_compat（OpenRouter 走 OpenAI 协议）
```

def generate_sql(...): ...

async def agenerate_sql(...): ...

async def _ainvoke_via_adapter(
    system_prompt: str, user_message: str,
    model_key: str, key: str, model_cfg: dict, provider: str,
    *, agent_kind: str = "sql_planner",   # R-26-Senior + R-32：必须传 agent_kind
):
    """v0.4.4 异步 LLM 调用入口。
    R-26-Senior：第一行先做 budget per_call check，早于 SDK 实例化（没钱不产生网络成本）
    R-32：agent_kind 默认 'sql_planner'，afix_sql 必须显式传 'fix_sql'
    """
    # R-26-Senior：budget block 在 LLM 请求之前
    estimated_cost = _estimate_cost(model_cfg)
    allowed, meta = budget_service.check_agent_per_call_budget(agent_kind, estimated_cost)
    if not allowed:
        raise BudgetExceededError(meta)  # models/errors.py 扩展（R-30）

    adapter = get_async_adapter(provider)
    try:
        resp = await adapter.acomplete(LLMRequest(
            model_key=model_key, system=system_prompt,
            messages=[{"role": "user", "content": user_message}],
            api_key=key, base_url=PROVIDER_BASE_URLS.get(provider, ""),
            max_tokens=MAX_TOKENS_PER_QUERY, temperature=SQL_TEMPERATURE,
        ))
    except Exception as e:
        # R-30 错误标准化：原 SDK 异常包装成领域异常
        msg = str(e)
        if "401" in msg or "authentication" in msg.lower():
            raise LLMAuthError(msg) from e
        if "429" in msg or "rate limit" in msg.lower():
            raise LLMRateLimitError(msg) from e
        raise LLMNetworkError(msg) from e

    cost = calculate_cost(resp.input_tokens, resp.output_tokens,
                          model_cfg.get("input_price", 0), model_cfg.get("output_price", 0))
    return resp.text, resp.input_tokens, resp.output_tokens, cost

### 6.2 `services/error_translator.py`（NEW）

```python
"""error_translator — BIAgentError → API 响应 dict 映射。

R-25：每个 ErrorKind 显式标 is_retryable，前端按此渲染重试按钮。
R-30：service 层职责（不放 models 层；models 仅定义形状）。
"""
from __future__ import annotations

from bi_agent.models.errors import (
    BIAgentError, BudgetExceededError, BusinessDBError,
    ConfigMissingError, DataSourceUnavailableError,
    LLMAuthError, LLMNetworkError, LLMRateLimitError, UnsafeSQLError,
)

# R-25：ErrorKind → user_message + is_retryable 映射表
_TRANSLATIONS = {
    BudgetExceededError:        {"kind": "budget_exceeded",
                                 "user_message": "已达预算上限，请联系管理员调整阈值",
                                 "is_retryable": False},   # R-25：严禁前端给重试按钮
    LLMAuthError:               {"kind": "config_missing",
                                 "user_message": "AI 服务认证失败，请联系管理员检查 API Key",
                                 "is_retryable": False},
    LLMRateLimitError:          {"kind": "llm_failed",
                                 "user_message": "AI 服务繁忙，请稍后重试",
                                 "is_retryable": True},
    LLMNetworkError:            {"kind": "llm_failed",
                                 "user_message": "AI 服务暂时无法响应，请稍后重试",
                                 "is_retryable": True},
    UnsafeSQLError:             {"kind": "sql_invalid",
                                 "user_message": "SQL 不安全或不符合规则（仅允许 SELECT）",
                                 "is_retryable": False},
    BusinessDBError:            {"kind": "sql_exec_failed",
                                 "user_message": "业务库执行失败，请稍后重试或联系管理员",
                                 "is_retryable": True},
    DataSourceUnavailableError: {"kind": "data_source_unavailable",
                                 "user_message": "数据库未配置或连接失败，请联系管理员",
                                 "is_retryable": True},
    ConfigMissingError:         {"kind": "config_missing",
                                 "user_message": "服务配置缺失（API Key 或模型），请联系管理员",
                                 "is_retryable": False},
}


def to_response(err: BIAgentError) -> dict:
    """统一 API 错误响应 shape。"""
    for cls, trans in _TRANSLATIONS.items():
        if isinstance(err, cls):
            return {
                **trans,
                "details": getattr(err, "meta", {}) or {"raw": str(err)[:200]},
            }
    # 兜底
    return {
        "kind": "internal",
        "user_message": "服务出错，请稍后再试",
        "is_retryable": True,
        "details": {"raw": str(err)[:200]},
    }
```
