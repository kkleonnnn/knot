    CompileError, _TIME_KEYS, _resolve_physical, _resolve_date_col, _json_list, _order_limit)

def build_multi_base_sql(lf, metrics_by_name, tables, time_ctx) -> str:
    """多 base 聚合分发（返 raw SQL；compiler 包 _guard）。无维度 → 标量；有维度 → 维度并集驱动。"""
