## 3. 技术选型

### 3.1 算法：Fernet（`cryptography>=41,<46`）
AES-128-CBC + HMAC-SHA256；自带 IV/timestamp/auth tag。

### 3.2 Master key：环境变量 `BIAGENT_MASTER_KEY`
- 唯一来源；不读 .env auto-load；不读 app_settings（避鸡生蛋）
- 缺失 / 格式错 → fail-fast，**严禁** silent fallback / 自动生成
- 未来 v0.5+ KMS 时统一加 file/KMS 双源

### 3.3 加密产物前缀：`enc_v1:`
- 写：`enc_v1:` + Fernet token
- 读：见前缀走 decrypt；无前缀视为老明文兼容（INFO log）
- 空字符串 → `enc_v1:` 占位（区分 NULL）

```python
# core/crypto/fernet.py
import os
from functools import lru_cache
from cryptography.fernet import Fernet, InvalidToken

ENC_PREFIX = "enc_v1:"
_ENV = "BIAGENT_MASTER_KEY"


class CryptoConfigError(Exception):
    """加密配置异常 — core 层本地（Contract 3 禁 core→models）；
    repositories catch 后翻译为 models.errors.ConfigMissingError。"""


class FernetAdapter:
    def __init__(self, master_key: bytes):
        self._fernet = Fernet(master_key)

    def encrypt(self, plaintext: str) -> str:
        if plaintext is None: return ""
        if plaintext == "": return ENC_PREFIX
        return ENC_PREFIX + self._fernet.encrypt(plaintext.encode("utf-8")).decode("ascii")

    def decrypt(self, ciphertext: str) -> str:
        if not ciphertext: return ""
        if not ciphertext.startswith(ENC_PREFIX):
            return ciphertext  # 老明文兼容（D5 INFO log 在 repos wrap 内打）
        body = ciphertext[len(ENC_PREFIX):]
        if body == "": return ""
        try:
            return self._fernet.decrypt(body.encode("ascii")).decode("utf-8")
        except InvalidToken as e:
            raise CryptoConfigError("解密失败 — master key 已变更或密文损坏") from e

    def is_encrypted(self, s: str) -> bool:
        return isinstance(s, str) and s.startswith(ENC_PREFIX)


@lru_cache(maxsize=1)
def get_crypto_adapter():
    """R-40 进程单例；测试 fixture cache_clear()。R-44 lru_cache 已线程安全。"""
    raw = os.environ.get(_ENV)
    if not raw:
        raise CryptoConfigError(f"{_ENV} 未设置 — v0.4.5 必须配置主密钥")
    try:
        return FernetAdapter(raw.encode() if isinstance(raw, str) else raw)
    except (ValueError, TypeError) as e:
        raise CryptoConfigError(f"{_ENV} 格式无效（需 Fernet base64-urlsafe 32 字节）: {e}") from e


def assert_master_key_loaded() -> None:
    get_crypto_adapter()
```

from bi_agent.core.crypto import encrypt, decrypt

from bi_agent.models.errors import ConfigMissingError

_USER_ENCRYPTED_COLS = ("api_key", "openrouter_api_key", "embedding_api_key", "doris_password")

def _decrypt_user_row(row):
    if row is None: return None
    out = dict(row)
    for col in _USER_ENCRYPTED_COLS:
        if col in out and out[col]:
            try: out[col] = decrypt(out[col])
            except CryptoConfigError as e:
                raise ConfigMissingError(str(e)) from e
    return out

### 4.4 迁移脚本（R-46 .bak + R-46-Tx 每表事务）

```python
"""bi_agent/scripts/migrate_encrypt_v045.py
执行：python3 -m bi_agent.scripts.migrate_encrypt_v045 [--dry-run]
"""
import argparse, shutil, sys
from pathlib import Path
from bi_agent.repositories.base import get_conn, DB_PATH
from bi_agent.core.crypto import encrypt, is_encrypted
from bi_agent.core.crypto.fernet import assert_master_key_loaded
from bi_agent.core.logging_setup import logger

TARGETS = [
    ("users", "id", ["api_key", "openrouter_api_key", "embedding_api_key", "doris_password"], None),
    ("data_sources", "id", ["db_password"], None),
    ("app_settings", "key", ["value"], "key IN ('openrouter_api_key', 'embedding_api_key')"),
]


def _backup(db_path: Path) -> Path:
    bak = db_path.with_suffix(db_path.suffix + ".v044.bak")
    if not bak.exists():
        shutil.copy2(db_path, bak)
        logger.info(f"[migrate] 已生成备份 {bak}")
    return bak


def migrate(dry_run: bool = False) -> dict:
    assert_master_key_loaded()
    if not dry_run: _backup(Path(DB_PATH))
    conn = get_conn()
    stats = {"scanned": 0, "encrypted": 0, "skipped": 0}
    for table, id_col, cols, where in TARGETS:
        sql = f"SELECT {id_col}, {','.join(cols)} FROM {table}"
        if where: sql += f" WHERE {where}"
        rows = conn.execute(sql).fetchall()
        try:
            for row in rows:
                stats["scanned"] += 1
                updates = {}
                for c in cols:
                    v = row[c]
                    if v is None or v == "" or is_encrypted(v): continue
                    updates[c] = encrypt(v)
                if updates and not dry_run:
                    set_clause = ",".join(f"{c}=?" for c in updates)
                    conn.execute(f"UPDATE {table} SET {set_clause} WHERE {id_col}=?",
                                 (*updates.values(), row[id_col]))
                    stats["encrypted"] += 1
                elif updates: stats["encrypted"] += 1
                else: stats["skipped"] += 1
            if not dry_run: conn.commit()
        except Exception:
            if not dry_run: conn.rollback()
            raise
    logger.info(f"[migrate] 完成: {stats}")
    return stats


if __name__ == "__main__":
    ap = argparse.ArgumentParser()
    ap.add_argument("--dry-run", action="store_true")
    sys.exit(0 if migrate(ap.parse_args().dry_run)["scanned"] >= 0 else 1)
```
