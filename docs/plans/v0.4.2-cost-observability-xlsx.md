### 3.5 `services/export_service.py` 加 xlsx（R-15 + R-S7）

```python
_XLSX_MAX_ROWS = 5000  # R-15 资深 Stage 2 硬限

def rows_to_xlsx_bytes(rows: list[dict], cols=None,
                       sheet_name: str = "Data") -> tuple[bytes, dict]:
    """R-15 + R-S7：5000 行硬限；返 (bytes, metadata) 把截断信息暴露给 API 层。
    metadata: {"truncated": bool, "total": int, "exported": int}"""
    from openpyxl import Workbook
    truncated = len(rows) > _XLSX_MAX_ROWS
    actual = rows[:_XLSX_MAX_ROWS] if truncated else rows
    cols = cols or (list(rows[0].keys()) if rows else [])
    wb = Workbook(); ws = wb.active; ws.title = sheet_name
    ws.append(cols)
    for r in actual:
        ws.append([_xlsx_stringify(r.get(c)) for c in cols])
    bio = BytesIO(); wb.save(bio)
    return bio.getvalue(), {
        "truncated": truncated,
        "total": len(rows),
        "exported": len(actual),
    }
```

API 层把截断信息放 response headers（**R-S7 强制**）：
```python
xlsx_bytes, meta = rows_to_xlsx_bytes(rows)
headers = {
    "Content-Disposition": f"attachment; filename*=UTF-8''{quote(filename)}",
    "X-Export-Truncated": str(meta["truncated"]).lower(),
    "X-Export-Total-Rows": str(meta["total"]),
    "X-Export-Returned-Rows": str(meta["exported"]),
}
```

前端导出按钮读响应 header：截断时 toast「已截断至 5000 行（共 N 行）。如需全量请联系 admin。」

### 5.2 守护正则升级（资深 Stage 2 AST hybrid 指导）

**优先 sqlglot AST 解析（已是依赖）+ 关键词正则双保险**：

```python
# tests/eval/test_eval.py 新加 dispatcher
def _check_complexity(case_id: str, sql: str, tags: list):
    """按 tag 调用对应守护检查（资深 AST hybrid 决议）。"""
    if "subquery" in tags: _check_subquery_present(case_id, sql)
    if "window" in tags:   _check_window_with_partition_or_order(case_id, sql)
    if "cte" in tags:      _check_cte_uses_with(case_id, sql)


def _check_window_with_partition_or_order(case_id: str, sql: str):
    """Window function: AST 检测 Window expression；fallback 正则 OVER (...)。"""
    sql_lower = sql.lower()
    # AST 优先
    try:
        import sqlglot
        from sqlglot import exp
        ast = sqlglot.parse_one(sql, dialect="mysql")
        windows = list(ast.find_all(exp.Window))
        if windows:
            for w in windows:
                if not (w.args.get("partition_by") or w.args.get("order")):
                    raise AssertionError(f"[{case_id}] window 缺 PARTITION BY 或 ORDER BY")
            return
    except Exception:
        pass
    # 关键词 fallback
    assert re.search(r"\bover\s*\(", sql_lower), f"[{case_id}] 缺 OVER (...)"
    assert re.search(r"\b(?:partition\s+by|order\s+by)\b", sql_lower), \
        f"[{case_id}] window 缺 PARTITION BY 或 ORDER BY"


def _check_cte_uses_with(case_id: str, sql: str):
    sql_lower = sql.strip().lower()
    assert sql_lower.startswith("with "), f"[{case_id}] CTE case 必须以 WITH 起手"
    # AST 验证
    try:
        import sqlglot
        from sqlglot import exp
        ast = sqlglot.parse_one(sql, dialect="mysql")
        ctes = list(ast.find_all(exp.CTE))
        assert len(ctes) >= 1, f"[{case_id}] AST 解析无 CTE"
    except Exception:
        pass


def _check_subquery_present(case_id: str, sql: str):
    """子查询 = SELECT 嵌套出现 ≥ 2 次（外层 + 子查询）。"""
    sql_lower = sql.lower()
    n = len(re.findall(r"\bselect\b", sql_lower))
    assert n >= 2, f"[{case_id}] subquery case 至少含 2 个 SELECT"
```
