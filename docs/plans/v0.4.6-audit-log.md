## 8. PII scrub 实现（R-48/59/62/D7 限深度 3）

```python
_MAX_SCRUB_DEPTH = 3  # D7 防恶意嵌套栈溢出

def _scrub_pii(d, _depth: int = 0):
    if d is None or _depth >= _MAX_SCRUB_DEPTH:
        return d
    if isinstance(d, dict):
        out = {}
        for k, v in d.items():
            if k in _PII_BLACKLIST:
                out[k] = "••••redacted••••"  # R-48 R-59 R-62
            elif isinstance(v, (dict, list)):
                out[k] = _scrub_pii(v, _depth + 1)
            else:
                out[k] = v
        return out
    if isinstance(d, list):
        return [_scrub_pii(x, _depth + 1) for x in d]
    return d
```

## 9. audit_service 入口（R-47/51/55/64）

# bi_agent/services/audit_service.py
def log(*, actor, action: AuditAction, resource_type, resource_id=None,
        success: bool = True, detail=None,
        client_ip=None, user_agent=None, request_id=None) -> None:
    """fire-and-log（R-47）：业务路径必须 try-free 调用。"""
    try:
        audit_repo.insert(
            actor_id=actor["id"] if actor else None,
            actor_role=actor["role"] if actor else None,
            actor_name=actor["username"] if actor else None,  # R-54 冗余快照
            action=action,
            resource_type=resource_type,
            resource_id=str(resource_id) if resource_id is not None else None,
            success=1 if success else 0,
            detail_json=_scrub_pii(detail) if detail else {},
            client_ip=client_ip,         # R-58 独立列
            user_agent=user_agent,       # R-58 独立列
            request_id=request_id,
        )
    except Exception as e:
        logger.error(f"[audit] 写入失败: {e}; action={action}")
        # R-64：prometheus counter hook 预埋（v0.4.6 还没接入，先注释 TODO + 全局变量）
        global _audit_write_failures_total
        _audit_write_failures_total += 1
