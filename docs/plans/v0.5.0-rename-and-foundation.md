def _read_master_key():
    new_key = os.getenv("KNOT_MASTER_KEY")
    old_key = os.getenv("BIAGENT_MASTER_KEY")

    # R-74 密文兼容性探针
    if new_key and old_key and new_key != old_key:
        # 找一条 enc_v1: 数据探针（users.api_key / openrouter_api_key / app_settings 任一非空行）
        # 试 old_key 解密成功而 new_key 失败 → sys.exit(1)
        if _try_decrypt_with(old_key) and not _try_decrypt_with(new_key):
            print("\033[1;31m...新 KEY 与历史加密数据不兼容...\033[0m", file=sys.stderr)
            sys.exit(1)

    if new_key:
        if old_key:
            logger.warning(
                "[deprecation] 同时设置了 KNOT_MASTER_KEY 和 BIAGENT_MASTER_KEY；"
                "使用 KNOT_MASTER_KEY，忽略 BIAGENT_MASTER_KEY。请删除旧变量。"
            )
        return new_key
    if old_key:
        logger.warning(
            "[deprecation] 仅检测到 BIAGENT_MASTER_KEY（v0.5.0 起改名为 KNOT_MASTER_KEY）；"
            "继续使用旧名以兼容；将于 v1.0 移除支持。"
        )
        return old_key
    return None

### 5.5 DB startup migration（F7）— 含 R-76 atomic + 双轨

`knot/scripts/migrate_db_rename_v050.py`（startup hook helper + 独立 entrypoint）：

```python
"""v0.5.0 DB rename migration（startup hook + 独立 entrypoint / 一次性 / 幂等 / atomic）。

执行：
    python3 -m knot.scripts.migrate_db_rename_v050 [--dry-run]

红线：
- R-69 幂等：4 场景（仅老 / 仅新 / 都无 / 都有）
- R-76 原子性：try/except 包裹；rename 失败 → 删 bak 保老 DB
"""
def migrate_db_rename(data_dir: Path) -> dict:
    old = data_dir / "bi_agent.db"
    new = data_dir / "knot.db"
    if not old.exists() and not new.exists():
        return {"status": "no_db_yet"}
    if not old.exists() and new.exists():
        return {"status": "already_migrated"}
    if old.exists() and new.exists():
        msg = f"[migrate-db-rename] 检测到 {old} 和 {new} 同时存在 — 请人工确认数据归属"
        logger.error(msg)
        raise RuntimeError(msg)

    ts = time.strftime("%Y%m%d-%H%M%S")
    bak = old.parent / f"{old.name}.v044-{ts}.bak"
    try:
        shutil.copy2(old, bak)
        old.rename(new)
    except OSError as e:
        # R-76 原子性：rename 失败时绝不允许 DB 消失
        if bak.exists() and not new.exists():
            bak.unlink()  # 避免误导用户以为已迁移
        logger.error(f"[migrate-db-rename] FAILED — old DB preserved: {old}")
        raise
    logger.info(f"[migrate-db-rename] {old} → {new}（备份保留 {bak}）")
    return {"status": "migrated", "backup": str(bak), "new_db": str(new)}
```

main.py 在 `init_db()` **之前**调用：

```python
from knot.scripts.migrate_db_rename_v050 import migrate_db_rename

(Path(__file__).parent / "data").mkdir(parents=True, exist_ok=True)
migrate_db_rename(Path(__file__).parent / "data")  # ← R-69 / R-76
init_db()
```
